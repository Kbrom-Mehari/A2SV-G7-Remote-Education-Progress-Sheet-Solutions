class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        n = len(arr)
        i = 0
        while i < n - 1:
            if arr[i] == 0:
                for j in range(n - 2, i - 1 , -1):
                    arr[j + 1] = arr[j]
                i += 2
            else:
                i += 1
